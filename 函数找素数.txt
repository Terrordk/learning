#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int is_prime(int n)
{
	int j = 0;
	for (j = 2; j < n; j++)
	{
		if (n % j == 0);
	}

	return 1;
}


int main()        //用函数的方式找出100-200之间的素数
{
	int i = 0;
	for (i < 100; i <= 200; i++)
	{
		if (is_prime(i) == 1)
		{
			printf("%d\n", i);
		}
		return 0;
	}
}
